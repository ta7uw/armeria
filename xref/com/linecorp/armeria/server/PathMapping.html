18  import static com.linecorp.armeria.internal.PathMappingUtil.EXACT;
19  import static com.linecorp.armeria.internal.PathMappingUtil.GLOB;
20  import static com.linecorp.armeria.internal.PathMappingUtil.PREFIX;
21  import static com.linecorp.armeria.internal.PathMappingUtil.REGEX;
22  import static java.util.Objects.requireNonNull;
23  
24  import java.util.List;
25  import java.util.Optional;
26  import java.util.Set;
27  import java.util.regex.Pattern;
28  
29  import com.google.common.collect.ImmutableList;
30  import com.google.common.collect.ImmutableSet;
31  
32  import com.linecorp.armeria.common.HttpMethod;
33  import com.linecorp.armeria.common.MediaType;
34  import com.linecorp.armeria.server.docs.DocService;
35  
36  import io.micrometer.core.instrument.Meter;
37  import io.micrometer.core.instrument.Tag;

39  /**
40   * Matches the absolute path part of a URI and extracts path parameters from it.
41   */
42  public interface PathMapping {
43  
44      /**
45       * Creates a new {@link PathMapping} that matches the specified {@code pathPattern}. e.g.
46       * <ul>
47       *   <li>{@code /login} (no path parameters)</li>
48       *   <li>{@code /users/{userId}} (curly-brace style)</li>
49       *   <li>{@code /list/:productType/by/:ordering} (colon style)</li>
50       *   <li>{@code exact:/foo/bar} (exact match)</li>
51       *   <li>{@code prefix:/files} (prefix match)</li>
52       *   <li><code>glob:/~&#42;/downloads/**</code> (glob pattern)</li>
53       *   <li>{@code regex:^/files/(?<filePath>.*)$} (regular expression)</li>
54       * </ul>
55       *
56       * @throws IllegalArgumentException if the specified path pattern is invalid
57       */
58      static PathMapping of(String pathPattern) {
59          requireNonNull(pathPattern, "pathPattern");
60  
61          if (pathPattern.startsWith(EXACT)) {
62              return ofExact(pathPattern.substring(EXACT.length()));
63          }
64          if (pathPattern.startsWith(PREFIX)) {
65              return ofPrefix(pathPattern.substring(PREFIX.length()));
66          }
67          if (pathPattern.startsWith(GLOB)) {
68              return ofGlob(pathPattern.substring(GLOB.length()));
69          }
70          if (pathPattern.startsWith(REGEX)) {
71              return ofRegex(pathPattern.substring(REGEX.length()));
72          }
73          if (!pathPattern.startsWith("/")) {
74              throw new IllegalArgumentException(
75                      "pathPattern: " + pathPattern + " (not an absolute path or a unknown pattern type)");
76          }
77          if (!pathPattern.contains("{") && !pathPattern.contains(":")) {
78              return ofExact(pathPattern);
79          }
80          return new DefaultPathMapping(pathPattern);
81      }
82  
83      /**
84       * Creates a new {@link PathMapping} that matches a {@linkplain ServiceRequestContext#path() path} with
85       * the specified regular expression and extracts the values of the named groups.
86       * e.g. {@code "^/users/(?<userId>[0-9]+)$"} will extract the second numeric part of the path into
87       *      the {@code "userId"} path parameter.
88       */
89      static PathMapping ofRegex(Pattern regex) {
90          return new RegexPathMapping(regex);
91      }
92  
93      /**
94       * Creates a new {@link PathMapping} that matches a {@linkplain ServiceRequestContext#path() path} with
95       * the specified regular expression and extracts the values of the named groups.
96       * e.g. {@code "^/users/(?<userId>[0-9]+)$"} will extract the second numeric part of the path into
97       *      the {@code "userId"} path parameter.
98       */
99      static PathMapping ofRegex(String regex) {
100         return ofRegex(Pattern.compile(requireNonNull(regex, "regex")));
101     }
102 
103     /**
104      * Creates a new {@link PathMapping} that matches a {@linkplain ServiceRequestContext#path() path} with
105      * the specified glob expression, where {@code "*"} matches a path component non-recursively and
106      * {@code "**"} matches path components recursively.
107      */
108     static PathMapping ofGlob(String glob) {
109         requireNonNull(glob, "glob");
110         if (glob.startsWith("/") && !glob.contains("*")) {
111             // Does not have a pattern matcher.
112             return ofExact(glob);
113         }
114 
115         return new GlobPathMapping(glob);
116     }
117 
118     /**
119      * Creates a new {@link PathMapping} that matches a {@linkplain ServiceRequestContext#path() path}
120      * under the specified directory prefix. It also removes the specified directory prefix from the matched
121      * path so that {@link ServiceRequestContext#path()} does not have the specified directory prefix.
122      * For example, when {@code pathPrefix} is {@code "/foo/"}:
123      * <ul>
124      *   <li>{@code "/foo/"} translates to {@code "/"}</li>
125      *   <li>{@code "/foo/bar"} translates to  {@code "/bar"}</li>
126      *   <li>{@code "/foo/bar/baz"} translates to {@code "/bar/baz"}</li>
127      * </ul>
128      * This method is a shortcut to {@link #ofPrefix(String, boolean) ofPrefix(pathPrefix, true)}.
129      */
130     static PathMapping ofPrefix(String pathPrefix) {
131         return ofPrefix(pathPrefix, true);
132     }
133 
134     /**
135      * Creates a new {@link PathMapping} that matches a {@linkplain ServiceRequestContext#path() path}
136      * under the specified directory prefix. When {@code stripPrefix} is {@code true}, it also removes the
137      * specified directory prefix from the matched path so that {@link ServiceRequestContext#path()}
138      * does not have the specified directory prefix. For example, when {@code pathPrefix} is {@code "/foo/"}:
139      * <ul>
140      *   <li>{@code "/foo/"} translates to {@code "/"}</li>
141      *   <li>{@code "/foo/bar"} translates to  {@code "/bar"}</li>
142      *   <li>{@code "/foo/bar/baz"} translates to {@code "/bar/baz"}</li>
143      * </ul>
144      */
145     static PathMapping ofPrefix(String pathPrefix, boolean stripPrefix) {
146         requireNonNull(pathPrefix, "pathPrefix");
147         if ("/".equals(pathPrefix)) {
148             // Every path starts with '/'.
149             return ofCatchAll();
150         }
151 
152         return new PrefixPathMapping(pathPrefix, stripPrefix);
153     }
154 
155     /**
156      * Creates a new {@link PathMapping} that performs an exact match.
157      */
158     static PathMapping ofExact(String exactPath) {
159         return new ExactPathMapping(exactPath);
160     }
161 
162     /**
163      * Returns a singleton {@link PathMapping} that always matches any path.
164      */
165     static PathMapping ofCatchAll() {
166         return CatchAllPathMapping.INSTANCE;
167     }
168 
169     /**
170      * Matches the specified {@code path} and extracts the path parameters from it.
171      *
172      * @param mappingCtx a context to find the {@link Service}.
173      *
174      * @return a non-empty {@link PathMappingResult} if the specified {@code path} matches this mapping.
175      *         {@link PathMappingResult#empty()} if not matches.
176      */
177     PathMappingResult apply(PathMappingContext mappingCtx);
178 
179     /**
180      * Returns the names of the path parameters extracted by this mapping.
181      */
182     Set<String> paramNames();
183 
184     /**
185      * Returns the logger name.
186      *
187      * @return the logger name whose components are separated by a dot (.)
188      */
189     String loggerName();
190 
191     /**
192      * Returns the value of the {@code "pathMapping"} {@link Tag} in a {@link Meter}.
193      */
194     String meterTag();
195 
196     /**
197      * Returns the exact path of this path mapping if it is an exact path mapping, or {@link Optional#empty}
198      * otherwise. This can be useful for services which provide logic after scanning the server's mapped
199      * services, e.g. {@link DocService}
200      */
201     Optional<String> exactPath();
202 
203     /**
204      * Returns the prefix of this path mapping if it is a prefix mapping or a prefix adding path mapping,
205      * {@link Optional#empty} otherwise. This can be useful for services which provide logic after scanning
206      * the server's mapped services, e.g. {@link DocService}
207      *
208      * @return the prefix that ends with '/' if this mapping is a prefix mapping
209      */
210     Optional<String> prefix();
211 
212     /**
213      * Returns the path of this path mapping if it is able to be added to a trie.
214      */
215     default Optional<String> triePath() {
216         return Optional.empty();
217     }
218 
219     /**
220      * Returns the regular expression of this path mapping if it is created with the regular expression or
221      * glob pattern. Please note that this regex does not include the {@code pathPrefix} if this is
222      * created with prefix adding path mapping. You should call {@link #prefix()} to retrieve that.
223      */
224     default Optional<String> regex() {
225         return Optional.empty();
226     }
227 
228     /**
229      * Returns the complexity of this path mapping. It would be increased if this path mapping has
230      * more conditions to check.
231      */
232     default int complexity() {
233         return 0;
234     }
235 
236     /**
237      * Returns the {@link Set} of {@link HttpMethod}s that this {@link PathMapping} supports.
238      */
239     default Set<HttpMethod> supportedMethods() {
240         return ImmutableSet.of();
241     }
242 
243     /**
244      * Returns the {@link List} of {@link MediaType}s that this {@link PathMapping} consumes.
245      */
246     default List<MediaType> consumeTypes() {
247         return ImmutableList.of();
248     }
249 
250     /**
251      * Returns the {@link List} of {@link MediaType}s that this {@link PathMapping} produces.
252      */
253     default List<MediaType> produceTypes() {
254         return ImmutableList.of();
255     }
256 
257     /**
258      * Creates a new {@link PathMapping} that matches a {@linkplain ServiceRequestContext#path() path} using
259      * the specified {@link PathMapping} and checks whether the request is valid using the specified
260      * {@code supportedMethods}, {@code consumeTypes} and {@code produceTypes}.
261      *
262      * @param supportedMethods the set of {@link HttpMethod}s that this {@link PathMapping} supports
263      * @param consumeTypes the list of {@link MediaType}s that this {@link PathMapping} consumes
264      * @param produceTypes the list of {@link MediaType}s that this {@link PathMapping} produces
265      */
266     default PathMapping withHttpHeaderInfo(Set<HttpMethod> supportedMethods,
267                                            List<MediaType> consumeTypes, List<MediaType> produceTypes) {
268         return new HttpHeaderPathMapping(this, supportedMethods, consumeTypes, produceTypes);
269     }
270 }
