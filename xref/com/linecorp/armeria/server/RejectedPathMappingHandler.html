16  package com.linecorp.armeria.server;
17  
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  /**
22   * A handler that is invoked when a {@link ServerBuilder} rejects to bind a {@link Service} at
23   * a certain {@link PathMapping}. For example, the following code will trigger this handler:
24   *
25   * <pre>{@code
26   * ServerBuilder sb = new ServerBuilder();
27   * sb.service("/hello", serviceA);
28   * sb.service("/hello", serviceB); // Tried to bind at the same path mapping again.
29   * sb.build();
30   * }</pre>
31   *
32   * @see ServerBuilder#rejectedPathMappingHandler(RejectedPathMappingHandler)
33   */
34  @FunctionalInterface
35  public interface RejectedPathMappingHandler {
36  
37      /**
38       * A {@link RejectedPathMappingHandler} that logs a warning message for a problematic {@link PathMapping}.
39       */
40      RejectedPathMappingHandler WARN = (virtualHost, mapping, existingMapping) -> {
41          final Logger logger = LoggerFactory.getLogger(RejectedPathMappingHandler.class);
42          final String a = mapping.toString();
43          final String b = existingMapping.toString();
44          final String hostnamePattern = virtualHost.hostnamePattern();
45          // TODO(trustin): Deduplicate message generation in 'WARN' and 'FAIL'
46          //                once we can have private methods in an interface.
47          if (a.equals(b)) {
48              logger.warn("Virtual host '{}' has a duplicate path mapping: {}",
49                          hostnamePattern, a);
50          } else {
51              logger.warn("Virtual host '{}' has path mappings with a conflict: {} vs. {}",
52                          hostnamePattern, a, b);
53          }
54      };
55  
56      /**
57       * A {@link RejectedPathMappingHandler} that raises an {@link IllegalStateException} for a problematic
58       * {@link PathMapping}.
59       */
60      RejectedPathMappingHandler FAIL = (virtualHost, mapping, existingMapping) -> {
61          final String a = mapping.toString();
62          final String b = existingMapping.toString();
63          final String hostnamePattern = virtualHost.hostnamePattern();
64          if (a.equals(b)) {
65              throw new IllegalStateException(
66                      "Virtual host '" + hostnamePattern + "' has a duplicate path mapping: " + a);
67          } else {
68              throw new IllegalStateException(
69                      "Virtual host '" + hostnamePattern + "' has path mappings with a conflict: " +
70                      a + " vs. " + b);
71          }
72      };
73  
74      /**
75       * Invoked when a user attempts to bind a {@link Service} at the {@link PathMapping} that conflicts with
76       * an existing {@link PathMapping}.
77       *
78       * @param virtualHost     the {@link VirtualHost} where the {@link PathMapping} belong to
79       * @param mapping         the {@link PathMapping} being added
80       * @param existingMapping the existing {@link PathMapping}
81       */
82      void handleDuplicatePathMapping(VirtualHost virtualHost,
83                                      PathMapping mapping, PathMapping existingMapping) throws Exception;
84  }
