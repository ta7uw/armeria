104     /**
105      * Creates a new instance.
106      *
107      * @param cfg the {@link ServiceConfig}
108      * @param ch the {@link Channel} that handles the invocation
109      * @param meterRegistry the {@link MeterRegistry} that collects various stats
110      * @param sessionProtocol the {@link SessionProtocol} of the invocation
111      * @param pathMappingContext the parameters which is used when finding a matched {@link PathMapping}
112      * @param pathMappingResult the result of finding a matched {@link PathMapping}
113      * @param request the request associated with this context
114      * @param sslSession the {@link SSLSession} for this invocation if it is over TLS
115      * @param proxiedAddresses source and destination addresses retrieved from PROXY protocol header
116      * @param clientAddress the address of a client who initiated the request
117      */
118     public DefaultServiceRequestContext(
119             ServiceConfig cfg, Channel ch, MeterRegistry meterRegistry, SessionProtocol sessionProtocol,
120             PathMappingContext pathMappingContext, PathMappingResult pathMappingResult, Request request,
121             @Nullable SSLSession sslSession, @Nullable ProxiedAddresses proxiedAddresses,
122             InetAddress clientAddress) {
123 
124         super(meterRegistry, sessionProtocol,
125               requireNonNull(pathMappingContext, "pathMappingContext").method(), pathMappingContext.path(),
126               requireNonNull(pathMappingResult, "pathMappingResult").query(),
127               request);
128 
129         this.ch = requireNonNull(ch, "ch");
130         this.cfg = requireNonNull(cfg, "cfg");
131         this.pathMappingContext = pathMappingContext;
132         this.pathMappingResult = pathMappingResult;
133         this.sslSession = sslSession;
134         this.proxiedAddresses = proxiedAddresses;
135         this.clientAddress = requireNonNull(clientAddress, "clientAddress");
136 
137         log = new DefaultRequestLog(this);
138         log.startRequest(ch, sessionProtocol);
139         logger = newLogger(cfg);
140 
141         final ServerConfig serverCfg = cfg.server().config();
142         requestTimeoutMillis = serverCfg.defaultRequestTimeoutMillis();
143         maxRequestLength = serverCfg.defaultMaxRequestLength();
144     }

146     private RequestContextAwareLogger newLogger(ServiceConfig cfg) {
147         String loggerName = cfg.loggerName().orElse(null);
148         if (loggerName == null) {
149             loggerName = cfg.pathMapping().loggerName();
150         }
151 
152         return new RequestContextAwareLogger(this, LoggerFactory.getLogger(
153                 cfg.server().config().serviceLoggerPrefix() + '.' + loggerName));
154     }

186     @Override
187     public ServiceRequestContext newDerivedContext(Request request) {
188         final DefaultServiceRequestContext ctx = new DefaultServiceRequestContext(
189                 cfg, ch, meterRegistry(), sessionProtocol(), pathMappingContext,
190                 pathMappingResult, request, sslSession(), proxiedAddresses(), clientAddress);
191 
192         final HttpHeaders additionalHeaders = additionalResponseHeaders();
193         if (!additionalHeaders.isEmpty()) {
194             ctx.setAdditionalResponseHeaders(additionalHeaders);
195         }
196 
197         final HttpHeaders additionalTrailers = additionalResponseTrailers();
198         if (!additionalTrailers.isEmpty()) {
199             ctx.setAdditionalResponseTrailers(additionalTrailers);
200         }
201 
202         for (final Iterator<Attribute<?>> i = attrs(); i.hasNext();/* noop */) {
203             ctx.addAttr(i.next());
204         }
205         return ctx;
206     }

208     private HttpHeaders createAdditionalHeadersIfAbsent() {
209         final HttpHeaders additionalResponseHeaders = this.additionalResponseHeaders;
210         if (additionalResponseHeaders == null) {
211             final HttpHeaders newHeaders = new DefaultHttpHeaders();
212             if (additionalResponseHeadersUpdater.compareAndSet(this, null, newHeaders)) {
213                 return newHeaders;
214             } else {
215                 final HttpHeaders additionalResponseHeadersSetByOtherThread = this.additionalResponseHeaders;
216                 assert additionalResponseHeadersSetByOtherThread != null;
217                 return additionalResponseHeadersSetByOtherThread;
218             }
219         } else {
220             return additionalResponseHeaders;
221         }
222     }
223 
224     private HttpHeaders createAdditionalTrailersIfAbsent() {
225         final HttpHeaders additionalResponseTrailers = this.additionalResponseTrailers;
226         if (additionalResponseTrailers == null) {
227             final HttpHeaders newTrailers = new DefaultHttpHeaders();
228             if (additionalResponseTrailersUpdater.compareAndSet(this, null, newTrailers)) {
229                 return newTrailers;
230             } else {
231                 final HttpHeaders additionalResponseTrailersSetByOtherThread = this.additionalResponseTrailers;
232                 assert additionalResponseTrailersSetByOtherThread != null;
233                 return additionalResponseTrailersSetByOtherThread;
234             }
235         } else {
236             return additionalResponseTrailers;
237         }
238     }

327     @Override
328     public void setRequestTimeoutMillis(long requestTimeoutMillis) {
329         if (requestTimeoutMillis < 0) {
330             throw new IllegalArgumentException(
331                     "requestTimeoutMillis: " + requestTimeoutMillis + " (expected: >= 0)");
332         }
333         if (this.requestTimeoutMillis != requestTimeoutMillis) {
334             this.requestTimeoutMillis = requestTimeoutMillis;
335             final RequestTimeoutChangeListener listener = requestTimeoutChangeListener;
336             if (listener != null) {
337                 if (ch.eventLoop().inEventLoop()) {
338                     listener.onRequestTimeoutChange(requestTimeoutMillis);
339                 } else {
340                     ch.eventLoop().execute(() -> listener.onRequestTimeoutChange(requestTimeoutMillis));
341                 }
342             }
343         }
344     }

478     /**
479      * Sets the listener that is notified when the {@linkplain #requestTimeoutMillis()} request timeout} of
480      * the request is changed.
481      *
482      * <p>Note: This method is meant for internal use by server-side protocol implementation to reschedule
483      * a timeout task when a user updates the request timeout configuration.
484      */
485     public void setRequestTimeoutChangeListener(RequestTimeoutChangeListener listener) {
486         requireNonNull(listener, "listener");
487         if (requestTimeoutChangeListener != null) {
488             throw new IllegalStateException("requestTimeoutChangeListener is set already.");
489         }
490         requestTimeoutChangeListener = listener;
491     }

493     @Override
494     public String toString() {
495         String strVal = this.strVal;
496         if (strVal != null) {
497             return strVal;
498         }
499 
500         final StringBuilder buf = new StringBuilder(96);
501 
502         // Prepend the current channel information if available.
503         final Channel ch = channel();
504         final boolean hasChannel = ch != null;
505         if (hasChannel) {
506             buf.append(ch);
507 
508             final InetAddress remote = ((InetSocketAddress) remoteAddress()).getAddress();
509             final InetAddress client = clientAddress();
510             if (remote != null && !remote.equals(client)) {
511                 buf.append("[C:").append(client.getHostAddress()).append(']');
512             }
513         }
514 
515         buf.append('[')
516            .append(sessionProtocol().uriText())
517            .append("://")
518            .append(virtualHost().defaultHostname());
519 
520         final InetSocketAddress laddr = localAddress();
521         if (laddr != null) {
522             buf.append(':').append(laddr.getPort());
523         } else {
524             buf.append(":-1"); // Port unknown.
525         }
526 
527         buf.append(path())
528            .append('#')
529            .append(method())
530            .append(']');
531 
532         strVal = buf.toString();
533         if (hasChannel) {
534             this.strVal = strVal;
535         }
536         return strVal;
537     }
