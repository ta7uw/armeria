238     static {
239         if (!isEpollAvailable()) {
240             final Throwable cause = Epoll.unavailabilityCause();
241             if (cause != null) {
242                 logger.info("/dev/epoll not available: {}", Exceptions.peel(cause).toString());
243             } else {
244                 if (HAS_WSLENV) {
245                     logger.info("/dev/epoll not available: WSL not supported");
246                 } else {
247                     logger.info("/dev/epoll not available: ?");
248                 }
249             }
250         } else if (USE_EPOLL) {
251             logger.info("Using /dev/epoll");
252         }
253 
254         if (!OpenSsl.isAvailable()) {
255             final Throwable cause = Exceptions.peel(OpenSsl.unavailabilityCause());
256             logger.info("OpenSSL not available: {}", cause.toString());
257         } else if (USE_OPENSSL) {
258             logger.info("Using OpenSSL: {}, 0x{}",
259                         OpenSsl.versionString(),
260                         Long.toHexString(OpenSsl.version() & 0xFFFFFFFFL));
261         }
262     }
263 
264     private static boolean isEpollAvailable() {
265         // Netty epoll transport does not work with WSL (Windows Sybsystem for Linux) yet.
266         // TODO(trustin): Re-enable on WSL if https://github.com/Microsoft/WSL/issues/1982 is resolved.
267         return Epoll.isAvailable() && !HAS_WSLENV;
268     }

671     private static Optional<String> caffeineSpec(String name, String defaultValue) {
672         final String spec = get(name, defaultValue, value -> {
673             try {
674                 if (!"off".equals(value)) {
675                     CaffeineSpec.parse(value);
676                 }
677                 return true;
678             } catch (Exception e) {
679                 return false;
680             }
681         });
682         return "off".equals(spec) ? Optional.empty()
683                                   : Optional.of(spec);
684     }

686     private static ExceptionVerbosity exceptionLoggingMode(String name, String defaultValue) {
687         final String mode = getNormalized(name, defaultValue,
688                                           value -> Arrays.stream(ExceptionVerbosity.values())
689                                                          .anyMatch(v -> v.name().equalsIgnoreCase(value)));
690         return ExceptionVerbosity.valueOf(mode.toUpperCase());

697     private static boolean getBoolean(String name, boolean defaultValue, Predicate<Boolean> validator) {
698         return "true".equals(getNormalized(name, String.valueOf(defaultValue), value -> {
699             if ("true".equals(value)) {
700                 return validator.test(true);
701             }
702 
703             if ("false".equals(value)) {
704                 return validator.test(false);
705             }

707             return false;
708         }));

733     private static String get(String name, String defaultValue, Predicate<String> validator) {
734         final String fullName = PREFIX + name;
735         final String value = System.getProperty(fullName);
736         if (value == null) {
737             logger.info("{}: {} (default)", fullName, defaultValue);
738             return defaultValue;
739         }
740 
741         if (validator.test(value)) {
742             logger.info("{}: {}", fullName, value);
743             return value;
744         }
745 
746         logger.info("{}: {} (default instead of: {})", fullName, defaultValue, value);
747         return defaultValue;
748     }

750     private static String getNormalized(String name, String defaultValue, Predicate<String> validator) {
751         final String fullName = PREFIX + name;
752         final String value = getLowerCased(fullName);
753         if (value == null) {
754             logger.info("{}: {} (default)", fullName, defaultValue);
755             return defaultValue;
756         }
757 
758         if (validator.test(value)) {
759             logger.info("{}: {}", fullName, value);
760             return value;
761         }
762 
763         logger.info("{}: {} (default instead of: {})", fullName, defaultValue, value);
764         return defaultValue;
765     }
766 
767     @Nullable
768     private static String getLowerCased(String fullName) {
769         String value = System.getProperty(fullName);
770         if (value != null) {
771             value = Ascii.toLowerCase(value);
772         }
773         return value;
774     }
