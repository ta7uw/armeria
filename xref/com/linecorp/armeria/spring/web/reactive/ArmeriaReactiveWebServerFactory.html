103     @Override
104     public WebServer getWebServer(HttpHandler httpHandler) {
105         final ServerBuilder sb = new ServerBuilder();
106 
107         final SessionProtocol protocol;
108         final Ssl ssl = getSsl();
109         if (ssl != null) {
110             if (ssl.isEnabled()) {
111                 configureTls(sb, ssl, getSslStoreProvider());
112                 protocol = SessionProtocol.HTTPS;
113             } else {
114                 logger.warn("TLS configuration exists but it is disabled by 'enabled' property.");
115                 protocol = SessionProtocol.HTTP;
116             }
117         } else {
118             protocol = SessionProtocol.HTTP;
119         }
120 
121         final Http2 http2 = getHttp2();
122         if (http2 != null && !http2.isEnabled()) {
123             logger.warn(
124                     "Cannot disable HTTP/2 protocol for Armeria server. It will be enabled automatically.");
125         }
126 
127         final InetAddress address = getAddress();
128         final int port = ensureValidPort(getPort());
129         if (address != null) {
130             sb.port(new InetSocketAddress(address, port), protocol);
131         } else {
132             sb.port(port, protocol);
133         }
134 
135         final Compression compression = getCompression();
136         if (compression != null && compression.getEnabled()) {
137             sb.decorator(contentEncodingDecorator(compression));
138         }
139 
140         findBean(ArmeriaSettings.class).ifPresent(settings -> configureArmeriaService(sb, settings));
141         findBeans(ArmeriaServerConfigurator.class).forEach(configurator -> configurator.configure(sb));
142 
143         final DataBufferFactoryWrapper<?> factoryWrapper =
144                 findBean(DataBufferFactoryWrapper.class).orElse(DataBufferFactoryWrapper.DEFAULT);
145         final Server server = configureService(sb, httpHandler, factoryWrapper, getServerHeader()).build();
146         return new ArmeriaWebServer(server, protocol, address, port);
147     }
148 
149     private static ServerBuilder configureService(ServerBuilder sb, HttpHandler httpHandler,
150                                                   DataBufferFactoryWrapper<?> factoryWrapper,
151                                                   @Nullable String serverHeader) {
152         final ArmeriaHttpHandlerAdapter handler =
153                 new ArmeriaHttpHandlerAdapter(httpHandler, factoryWrapper);
154         return sb.service(PathMapping.ofCatchAll(), (ctx, req) -> {
155             final CompletableFuture<HttpResponse> future = new CompletableFuture<>();
156             final HttpResponse response = HttpResponse.from(future);
157             final Disposable disposable = handler.handle(ctx, req, future, serverHeader).subscribe();
158             response.completionFuture().whenComplete((unused, cause) -> {
159                 if (cause != null) {
160                     logger.debug("{} Response stream has been cancelled.", ctx, cause);
161                     disposable.dispose();
162                 }
163             });
164             return response;
165         });
166     }

168     private static Function<Service<HttpRequest, HttpResponse>,
169             HttpEncodingService> contentEncodingDecorator(Compression compression) {
170         final Predicate<MediaType> encodableContentTypePredicate;
171         final String[] mimeTypes = compression.getMimeTypes();
172         if (mimeTypes == null || mimeTypes.length == 0) {
173             encodableContentTypePredicate = contentType -> true;
174         } else {
175             final List<MediaType> encodableContentTypes =
176                     Arrays.stream(mimeTypes).map(MediaType::parse).collect(toImmutableList());
177             encodableContentTypePredicate = contentType ->
178                     encodableContentTypes.stream().anyMatch(contentType::is);
179         }
180 
181         final Predicate<HttpHeaders> encodableRequestHeadersPredicate;
182         final String[] excludedUserAgents = compression.getExcludedUserAgents();
183         if (excludedUserAgents == null || excludedUserAgents.length == 0) {
184             encodableRequestHeadersPredicate = headers -> true;
185         } else {
186             final List<Pattern> patterns =
187                     Arrays.stream(excludedUserAgents).map(Pattern::compile).collect(toImmutableList());
188             encodableRequestHeadersPredicate = headers -> {
189                 // No User-Agent header will be converted to an empty string.
190                 final String userAgent = headers.get(HttpHeaderNames.USER_AGENT, "");
191                 return patterns.stream().noneMatch(pattern -> pattern.matcher(userAgent).matches());
192             };
193         }
194 
195         return delegate -> new HttpEncodingService(delegate,
196                                                    encodableContentTypePredicate,
197                                                    encodableRequestHeadersPredicate,
198                                                    compression.getMinResponseSize().toBytes());
199     }

201     private void configureArmeriaService(ServerBuilder sb, ArmeriaSettings settings) {
202         final MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory =
203                 settings.isEnableMetrics() ? findBean(MeterIdPrefixFunctionFactory.class).orElse(DEFAULT)
204                                            : null;
205 
206         configurePorts(sb, settings.getPorts());
207         configureThriftServices(sb,
208                                 findBeans(ThriftServiceRegistrationBean.class),
209                                 meterIdPrefixFunctionFactory,
210                                 settings.getDocsPath());
211         configureHttpServices(sb,
212                               findBeans(HttpServiceRegistrationBean.class),
213                               meterIdPrefixFunctionFactory);
214         configureAnnotatedHttpServices(sb,
215                                        findBeans(AnnotatedServiceRegistrationBean.class),
216                                        meterIdPrefixFunctionFactory);
217         configureServerWithArmeriaSettings(sb, settings,
218                                            findBean(MeterRegistry.class).orElse(Metrics.globalRegistry),
219                                            findBeans(HealthChecker.class));
220     }

222     private <T> Optional<T> findBean(Class<T> clazz) {
223         final List<T> beans = findBeans(clazz);
224         switch (beans.size()) {
225             case 0:
226                 return Optional.empty();
227             case 1:
228                 return Optional.of(beans.get(0));
229             default:
230                 throw new IllegalStateException("Too many " + clazz.getSimpleName() + " beans: " +
231                                                 beans + " (expected: 1)");
232         }
233     }
234 
235     private <T> List<T> findBeans(Class<T> clazz) {
236         final String[] names = beanFactory.getBeanNamesForType(clazz);
237         if (names.length == 0) {
238             return ImmutableList.of();
239         }
240         return Arrays.stream(names)
241                      .map(name -> beanFactory.getBean(name, clazz))
242                      .collect(toImmutableList());
243     }

245     private static int ensureValidPort(int port) {
246         checkArgument(port >= 0 && port <= 65535,
247                       "port: %s (expected: 0[arbitrary port] or 1-65535)", port);
248         return port;
249     }

251     private static void configureTls(ServerBuilder sb,
252                                      Ssl ssl, @Nullable SslStoreProvider sslStoreProvider) {
253         try {
254             if (sslStoreProvider == null &&
255                 ssl.getKeyStore() == null && ssl.getTrustStore() == null) {
256                 logger.warn("Configuring TLS with a self-signed certificate " +
257                             "because no key or trust store was specified");
258                 sb.tlsSelfSigned();
259                 return;
260             }
261 
262             final SslContextBuilder sslBuilder = SslContextBuilder
263                     .forServer(getKeyManagerFactory(ssl, sslStoreProvider))
264                     .trustManager(getTrustManagerFactory(ssl, sslStoreProvider));
265 
266             final String[] enabledProtocols = ssl.getEnabledProtocols();
267             if (enabledProtocols != null) {
268                 sslBuilder.protocols(enabledProtocols.clone());
269             }
270 
271             final String[] ciphers = ssl.getCiphers();
272             if (ciphers != null) {
273                 sslBuilder.ciphers(ImmutableList.copyOf(ciphers));
274             }
275 
276             final ClientAuth clientAuth = ssl.getClientAuth();
277             if (clientAuth != null) {
278                 switch (clientAuth) {
279                     case NEED:
280                         sslBuilder.clientAuth(io.netty.handler.ssl.ClientAuth.REQUIRE);
281                         break;
282                     case WANT:
283                         sslBuilder.clientAuth(io.netty.handler.ssl.ClientAuth.OPTIONAL);
284                         break;
285                 }
286             }
287 
288             sb.tls(sslBuilder.build());
289         } catch (Exception e) {
290             throw new IllegalStateException("Failed to configure TLS: " + e);
291         }
292     }

294     private static KeyManagerFactory getKeyManagerFactory(
295             Ssl ssl, @Nullable SslStoreProvider sslStoreProvider) throws Exception {
296         final KeyStore store;
297         if (sslStoreProvider != null) {
298             store = sslStoreProvider.getKeyStore();
299         } else {
300             store = loadKeyStore(ssl.getKeyStoreType(), ssl.getKeyStore(), ssl.getKeyStorePassword());
301         }
302 
303         final KeyManagerFactory keyManagerFactory =
304                 KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
305 
306         String keyPassword = ssl.getKeyPassword();
307         if (keyPassword == null) {
308             keyPassword = ssl.getKeyStorePassword();
309         }
310 
311         keyManagerFactory.init(store, keyPassword != null ? keyPassword.toCharArray()
312                                                           : null);
313         return keyManagerFactory;
314     }
315 
316     private static TrustManagerFactory getTrustManagerFactory(
317             Ssl ssl, @Nullable SslStoreProvider sslStoreProvider) throws Exception {
318         final KeyStore store;
319         if (sslStoreProvider != null) {
320             store = sslStoreProvider.getTrustStore();
321         } else {
322             store = loadKeyStore(ssl.getTrustStoreType(), ssl.getTrustStore(), ssl.getTrustStorePassword());
323         }
324 
325         final TrustManagerFactory trustManagerFactory =
326                 TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
327         trustManagerFactory.init(store);
328         return trustManagerFactory;
329     }
330 
331     @Nullable
332     private static KeyStore loadKeyStore(
333             @Nullable String type,
334             @Nullable String resource,
335             @Nullable String password) throws IOException, GeneralSecurityException {
336         if (resource == null) {
337             return null;
338         }
339         final KeyStore store = KeyStore.getInstance(firstNonNull(type, "JKS"));
340         final URL url = ResourceUtils.getURL(resource);
341         store.load(url.openStream(), password != null ? password.toCharArray()
342                                                       : null);
343         return store;
344     }
