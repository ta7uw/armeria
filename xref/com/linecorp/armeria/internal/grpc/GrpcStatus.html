53  /**
54   * Utilities for handling {@link Status} in Armeria.
55   */
56  public final class GrpcStatus {
57  
58      /**
59       * Converts the {@link Throwable} to a {@link Status}, taking into account exceptions specific to Armeria as
60       * well.
61       */
62      public static Status fromThrowable(Throwable t) {
63          requireNonNull(t, "t");
64          final Status s = Status.fromThrowable(t);
65          if (s.getCode() != Code.UNKNOWN) {
66              return s;
67          }
68          if (t instanceof StreamException) {
69              final StreamException streamException = (StreamException) t;
70              if (streamException.getMessage() != null && streamException.getMessage().contains("RST_STREAM")) {
71                  return Status.CANCELLED;
72              }
73          }
74          if (t instanceof ClosedChannelException) {
75              // ClosedChannelException is used any time the Netty channel is closed. Proper error
76              // processing requires remembering the error that occurred before this one and using it
77              // instead.
78              return Status.UNKNOWN.withCause(t);
79          }
80          if (t instanceof IOException) {
81              return Status.UNAVAILABLE.withCause(t);
82          }
83          if (t instanceof Http2Exception) {
84              return Status.INTERNAL.withCause(t);
85          }
86          if (t instanceof TimeoutException) {
87              return Status.DEADLINE_EXCEEDED.withCause(t);
88          }
89          return s;
90      }
91  
92      /**
93       * Maps GRPC status codes to http status, as defined in upstream grpc-gateway
94       * <a href="https://github.com/grpc-ecosystem/grpc-gateway/blob/master/third_party/googleapis/google/rpc/code.proto">code.proto</a>.
95       */
96      public static HttpStatus grpcCodeToHttpStatus(Status.Code grpcStatusCode) {
97          switch (grpcStatusCode) {
98              case OK:
99                  return HttpStatus.OK;
100             case CANCELLED:
101                 return HttpStatus.CLIENT_CLOSED_REQUEST;
102             case UNKNOWN:
103             case INTERNAL:
104             case DATA_LOSS:
105                 return HttpStatus.INTERNAL_SERVER_ERROR;
106             case INVALID_ARGUMENT:
107             case FAILED_PRECONDITION:
108             case OUT_OF_RANGE:
109                 return HttpStatus.BAD_REQUEST;
110             case DEADLINE_EXCEEDED:
111                 return HttpStatus.GATEWAY_TIMEOUT;
112             case NOT_FOUND:
113                 return HttpStatus.NOT_FOUND;
114             case ALREADY_EXISTS:
115             case ABORTED:
116                 return HttpStatus.CONFLICT;
117             case PERMISSION_DENIED:
118                 return HttpStatus.FORBIDDEN;
119             case UNAUTHENTICATED:
120                 return HttpStatus.UNAUTHORIZED;
121             case RESOURCE_EXHAUSTED:
122                 return HttpStatus.TOO_MANY_REQUESTS;
123             case UNIMPLEMENTED:
124                 return HttpStatus.NOT_IMPLEMENTED;
125             case UNAVAILABLE:
126                 return HttpStatus.SERVICE_UNAVAILABLE;
127             default:
128                 return HttpStatus.UNKNOWN;
129         }
130     }
131 
132     /**
133      * Maps HTTP error response status codes to transport codes, as defined in <a
134      * href="https://github.com/grpc/grpc/blob/master/doc/http-grpc-status-mapping.md">
135      * http-grpc-status-mapping.md</a>. Never returns a status for which {@code status.isOk()} is
136      * {@code true}.
137      *
138      * <p>Copied from <a href="https://github.com/grpc/grpc-java/blob/master/core/src/main/java/io/grpc/internal/GrpcUtil.java">
139      * GrpcUtil.java</a>
140      */
141     public static Status httpStatusToGrpcStatus(int httpStatusCode) {
142         return httpStatusToGrpcCode(httpStatusCode).toStatus()
143                                                    .withDescription("HTTP status code " + httpStatusCode);
144     }
145 
146     private static Status.Code httpStatusToGrpcCode(int httpStatusCode) {
147         if (httpStatusCode >= 100 && httpStatusCode < 200) {
148             // 1xx. These headers should have been ignored.
149             return Status.Code.INTERNAL;
150         }
151         switch (httpStatusCode) {
152             case HttpURLConnection.HTTP_BAD_REQUEST:  // 400
153             case 431: // Request Header Fields Too Large
154                 return Status.Code.INTERNAL;
155             case HttpURLConnection.HTTP_UNAUTHORIZED:  // 401
156                 return Status.Code.UNAUTHENTICATED;
157             case HttpURLConnection.HTTP_FORBIDDEN:  // 403
158                 return Status.Code.PERMISSION_DENIED;
159             case HttpURLConnection.HTTP_NOT_FOUND:  // 404
160                 return Status.Code.UNIMPLEMENTED;
161             case 429:  // Too Many Requests
162             case HttpURLConnection.HTTP_BAD_GATEWAY:  // 502
163             case HttpURLConnection.HTTP_UNAVAILABLE:  // 503
164             case HttpURLConnection.HTTP_GATEWAY_TIMEOUT:  // 504
165                 return Status.Code.UNAVAILABLE;
166             default:
167                 return Status.Code.UNKNOWN;
168         }
169     }
170 
171     /**
172      * Fills the information from the {@link Throwable} into a {@link ThrowableProto} for
173      * returning to a client.
174      */
175     public static ThrowableProto serializeThrowable(Throwable t) {
176         final ThrowableProto.Builder builder = ThrowableProto.newBuilder();
177 
178         if (t instanceof StatusCauseException) {
179             final StatusCauseException statusCause = (StatusCauseException) t;
180             builder.setOriginalClassName(statusCause.getOriginalClassName());
181             builder.setOriginalMessage(statusCause.getOriginalMessage());
182         } else {
183             builder.setOriginalClassName(t.getClass().getCanonicalName());
184             builder.setOriginalMessage(Strings.nullToEmpty(t.getMessage()));
185         }
186 
187         for (StackTraceElement element : t.getStackTrace()) {
188             builder.addStackTrace(serializeStackTraceElement(element));
189         }
190 
191         if (t.getCause() != null) {
192             builder.setCause(serializeThrowable(t.getCause()));
193         }
194         return builder.build();
195     }
196 
197     private static StackTraceElementProto serializeStackTraceElement(StackTraceElement element) {
198         final StackTraceElementProto.Builder builder = StackTraceElementProto.newBuilder()
199                 .setClassName(element.getClassName())
200                 .setMethodName(element.getMethodName())
201                 .setLineNumber(element.getLineNumber());
202         if (element.getFileName() != null) {
203             builder.setFileName(element.getFileName());
204         }
205         return builder.build();
206     }
