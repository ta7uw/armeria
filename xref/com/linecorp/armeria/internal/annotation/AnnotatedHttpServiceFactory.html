140     /**
141      * Mapping from HTTP method annotation to {@link HttpMethod}, like following.
142      * <ul>
143      *   <li>{@link Options} -> {@link HttpMethod#OPTIONS}
144      *   <li>{@link Get} -> {@link HttpMethod#GET}
145      *   <li>{@link Head} -> {@link HttpMethod#HEAD}
146      *   <li>{@link Post} -> {@link HttpMethod#POST}
147      *   <li>{@link Put} -> {@link HttpMethod#PUT}
148      *   <li>{@link Patch} -> {@link HttpMethod#PATCH}
149      *   <li>{@link Delete} -> {@link HttpMethod#DELETE}
150      *   <li>{@link Trace} -> {@link HttpMethod#TRACE}
151      * </ul>
152      */
153     private static final Map<Class<?>, HttpMethod> HTTP_METHOD_MAP =
154             ImmutableMap.<Class<?>, HttpMethod>builder()
155                     .put(Options.class, HttpMethod.OPTIONS)
156                     .put(Get.class, HttpMethod.GET)
157                     .put(Head.class, HttpMethod.HEAD)
158                     .put(Post.class, HttpMethod.POST)
159                     .put(Put.class, HttpMethod.PUT)
160                     .put(Patch.class, HttpMethod.PATCH)
161                     .put(Delete.class, HttpMethod.DELETE)
162                     .put(Trace.class, HttpMethod.TRACE)
163                     .build();

165     /**
166      * Returns the list of {@link AnnotatedHttpService} defined by {@link Path} and HTTP method annotations
167      * from the specified {@code object}.
168      */
169     public static List<AnnotatedHttpServiceElement> find(String pathPrefix, Object object,
170                                                          Iterable<?> exceptionHandlersAndConverters) {
171         Builder<ExceptionHandlerFunction> exceptionHandlers = null;
172         Builder<RequestConverterFunction> requestConverters = null;
173         Builder<ResponseConverterFunction> responseConverters = null;
174 
175         for (final Object o : exceptionHandlersAndConverters) {
176             boolean added = false;
177             if (o instanceof ExceptionHandlerFunction) {
178                 if (exceptionHandlers == null) {
179                     exceptionHandlers = ImmutableList.builder();
180                 }
181                 exceptionHandlers.add((ExceptionHandlerFunction) o);
182                 added = true;
183             }
184             if (o instanceof RequestConverterFunction) {
185                 if (requestConverters == null) {
186                     requestConverters = ImmutableList.builder();
187                 }
188                 requestConverters.add((RequestConverterFunction) o);
189                 added = true;
190             }
191             if (o instanceof ResponseConverterFunction) {
192                 if (responseConverters == null) {
193                     responseConverters = ImmutableList.builder();
194                 }
195                 responseConverters.add((ResponseConverterFunction) o);
196                 added = true;
197             }
198             if (!added) {
199                 throw new IllegalArgumentException(o.getClass().getName() +
200                                                    " is neither an exception handler nor a converter.");
201             }
202         }
203 
204         final List<ExceptionHandlerFunction> exceptionHandlerFunctions =
205                 exceptionHandlers != null ? exceptionHandlers.build() : ImmutableList.of();
206         final List<RequestConverterFunction> requestConverterFunctions =
207                 requestConverters != null ? requestConverters.build() : ImmutableList.of();
208         final List<ResponseConverterFunction> responseConverterFunctions =
209                 responseConverters != null ? responseConverters.build() : ImmutableList.of();
210 
211         final List<Method> methods = requestMappingMethods(object);
212         return methods.stream()
213                       .map((Method method) -> create(pathPrefix, object, method, exceptionHandlerFunctions,
214                                                      requestConverterFunctions, responseConverterFunctions))
215                       .collect(toImmutableList());
216     }

218     /**
219      * Returns an {@link AnnotatedHttpService} instance defined to {@code method} of {@code object} using
220      * {@link Path} annotation.
221      */
222     private static AnnotatedHttpServiceElement create(String pathPrefix, Object object, Method method,
223                                                       List<ExceptionHandlerFunction> baseExceptionHandlers,
224                                                       List<RequestConverterFunction> baseRequestConverters,
225                                                       List<ResponseConverterFunction> baseResponseConverters) {
226 
227         final Set<Annotation> methodAnnotations = httpMethodAnnotations(method);
228         if (methodAnnotations.isEmpty()) {
229             throw new IllegalArgumentException("HTTP Method specification is missing: " + method.getName());
230         }
231 
232         final Set<HttpMethod> methods = toHttpMethods(methodAnnotations);
233         if (methods.isEmpty()) {
234             throw new IllegalArgumentException(method.getDeclaringClass().getName() + '#' + method.getName() +
235                                                " must have an HTTP method annotation.");
236         }
237 
238         final Class<?> clazz = object.getClass();
239         final PathMapping pathMapping = pathStringMapping(pathPrefix, method, methodAnnotations)
240                 .withHttpHeaderInfo(methods, consumableMediaTypes(method, clazz),
241                                     producibleMediaTypes(method, clazz));
242 
243         final List<ExceptionHandlerFunction> eh =
244                 exceptionHandlers(method, clazz).addAll(baseExceptionHandlers)
245                                                 .add(defaultExceptionHandler).build();
246         final List<RequestConverterFunction> req =
247                 requestConverters(method, clazz).addAll(baseRequestConverters).build();
248         final List<ResponseConverterFunction> res =
249                 responseConverters(method, clazz).addAll(baseResponseConverters).build();
250 
251         List<AnnotatedValueResolver> resolvers;
252         try {
253             resolvers = AnnotatedValueResolver.ofServiceMethod(method, pathMapping.paramNames(),
254                                                                toRequestObjectResolvers(req));
255         } catch (NoParameterException ignored) {
256             // Allow no parameter like below:
257             //
258             // @Get("/")
259             // public String method1() { ... }
260             //
261             resolvers = ImmutableList.of();
262         }
263 
264         final Set<String> expectedParamNames = pathMapping.paramNames();
265         final Set<String> requiredParamNames =
266                 resolvers.stream()
267                          .filter(AnnotatedValueResolver::isPathVariable)
268                          .map(AnnotatedValueResolver::httpElementName)
269                          .collect(Collectors.toSet());
270 
271         if (!expectedParamNames.containsAll(requiredParamNames)) {
272             final Set<String> missing = Sets.difference(requiredParamNames, expectedParamNames);
273             throw new IllegalArgumentException("cannot find path variables: " + missing);
274         }
275 
276         // Warn unused path variables only if there's no '@RequestObject' annotation.
277         if (resolvers.stream().noneMatch(r -> r.annotationType() == RequestObject.class) &&
278             !requiredParamNames.containsAll(expectedParamNames)) {
279             final Set<String> missing = Sets.difference(expectedParamNames, requiredParamNames);
280             logger.warn("Some path variables of the method '" + method.getName() +
281                         "' of the class '" + clazz.getName() +
282                         "' do not have their corresponding parameters annotated with @Param. " +
283                         "They would not be automatically injected: " + missing);
284         }
285 
286         final Optional<HttpStatus> defaultResponseStatus = findAnnotation(method, StatusCode.class)
287                 .map(code -> {
288                     final int statusCode = code.value();
289                     checkArgument(statusCode >= 0,
290                                   "invalid HTTP status code: %s (expected: >= 0)", statusCode);
291                     return HttpStatus.valueOf(statusCode);
292                 });
293 
294         // A CORS preflight request can be received because we handle it specially. The following
295         // decorator will prevent the service from an unexpected request which has OPTIONS method.
296         final Function<Service<HttpRequest, HttpResponse>,
297                 ? extends Service<HttpRequest, HttpResponse>> initialDecorator;
298         if (methods.contains(HttpMethod.OPTIONS)) {
299             initialDecorator = Function.identity();
300         } else {
301             initialDecorator = delegate -> new SimpleDecoratingService<HttpRequest, HttpResponse>(delegate) {
302                 @Override
303                 public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
304                     if (req.method() == HttpMethod.OPTIONS) {
305                         // This must be a CORS preflight request.
306                         throw HttpStatusException.of(HttpStatus.FORBIDDEN);
307                     }
308                     return delegate().serve(ctx, req);
309                 }
310             };
311         }
312         return new AnnotatedHttpServiceElement(pathMapping,
313                                                new AnnotatedHttpService(object, method, resolvers, eh,
314                                                                         res, pathMapping,
315                                                                         defaultResponseStatus),
316                                                decorator(method, clazz, initialDecorator));
317     }

319     /**
320      * Returns the list of {@link Path} annotated methods.
321      */
322     private static List<Method> requestMappingMethods(Object object) {
323         return getAllMethods(object.getClass(), withModifier(Modifier.PUBLIC))
324                 .stream()
325                 .filter(m -> getAllAnnotations(m).stream()
326                                                  .map(Annotation::annotationType)
327                                                  .anyMatch(a -> a == Path.class ||
328                                                                 HTTP_METHOD_MAP.containsKey(a)))
329                 .sorted(Comparator.comparingInt(AnnotatedHttpServiceFactory::order))
330                 .collect(toImmutableList());
331     }

389     private static List<MediaType> consumableMediaTypes(AnnotatedElement element) {
390         final List<MediaType> mediaTypes = new ArrayList<>();
391 
392         for (final Annotation annotation : getAllAnnotations(element)) {
393             if (annotation instanceof ConsumesGroup) {
394                 Arrays.stream(((ConsumesGroup) annotation).value())
395                       .forEach(e -> addConsumableMediaType(mediaTypes, MediaType.parse(e.value())));
396             } else if (annotation instanceof Consumes) {
397                 addConsumableMediaType(mediaTypes, MediaType.parse(((Consumes) annotation).value()));
398             } else if (annotation instanceof ConsumeTypes) {
399                 Arrays.stream(((ConsumeTypes) annotation).value())
400                       .forEach(e -> addConsumableMediaType(mediaTypes, MediaType.parse(e.value())));
401             } else if (annotation instanceof ConsumeType) {
402                 addConsumableMediaType(mediaTypes, MediaType.parse(((ConsumeType) annotation).value()));
403             } else {
404                 findAnnotations(annotation.annotationType(), Consumes.class)
405                         .forEach(e -> addConsumableMediaType(mediaTypes, MediaType.parse(e.value())));
406             }
407         }
408         return mediaTypes;
409     }

423     private static List<MediaType> producibleMediaTypes(AnnotatedElement element) {
424         final List<MediaType> mediaTypes = new ArrayList<>();
425 
426         for (final Annotation annotation : getAllAnnotations(element)) {
427             if (annotation instanceof ProducesGroup) {

430             } else if (annotation instanceof Produces) {
431                 addProducibleMediaType(mediaTypes, MediaType.parse(((Produces) annotation).value()));
432             } else if (annotation instanceof ProduceTypes) {
433                 Arrays.stream(((ProduceTypes) annotation).value())
434                       .forEach(e -> addProducibleMediaType(mediaTypes, MediaType.parse(e.value())));
435             } else if (annotation instanceof ProduceType) {
436                 addProducibleMediaType(mediaTypes, MediaType.parse(((ProduceType) annotation).value()));
437             } else {
438                 findAnnotations(annotation.annotationType(), Produces.class)
439                         .forEach(e -> addProducibleMediaType(mediaTypes, MediaType.parse(e.value())));
440             }
441         }
442         return mediaTypes;
443     }

449     private static void addMediaType(List<MediaType> mediaTypes, MediaType newMediaType,
450                                      Class<?> clazz, boolean allowWildcard) {
451         if (!allowWildcard && newMediaType.hasWildcard()) {
452             throw new IllegalArgumentException('@' + clazz.getSimpleName() + " must not have a wildcard: " +
453                                                newMediaType);
454         }
455         if (mediaTypes.stream().anyMatch(e -> e.equals(newMediaType))) {
456             throw new IllegalArgumentException("Duplicated media type for @" + clazz.getSimpleName() + ": " +
457                                                newMediaType);
458         }
459         mediaTypes.add(newMediaType);
460     }

462     /**
463      * Returns the {@link PathMapping} instance mapped to {@code method}.
464      */
465     private static PathMapping pathStringMapping(String pathPrefix, Method method,
466                                                  Set<Annotation> methodAnnotations) {
467         pathPrefix = ensureAbsolutePath(pathPrefix, "pathPrefix");
468         if (!pathPrefix.endsWith("/")) {
469             pathPrefix += '/';
470         }
471 
472         final String pattern = findPattern(method, methodAnnotations);
473         final PathMapping mapping = PathMapping.of(pattern);
474         if ("/".equals(pathPrefix)) {
475             // pathPrefix is not specified or "/".
476             return mapping;
477         }
478 
479         if (pattern.startsWith(EXACT)) {
480             return PathMapping.ofExact(concatPaths(
481                     pathPrefix, pattern.substring(EXACT.length())));
482         }
483 
484         if (pattern.startsWith(PREFIX)) {
485             return PathMapping.ofPrefix(concatPaths(
486                     pathPrefix, pattern.substring(PREFIX.length())));
487         }
488 
489         if (pattern.startsWith(GLOB)) {
490             final String glob = pattern.substring(GLOB.length());
491             if (glob.startsWith("/")) {
492                 return PathMapping.ofGlob(concatPaths(pathPrefix, glob));
493             } else {
494                 // NB: We cannot use PathMapping.ofGlob(pathPrefix + "/**/" + glob) here
495                 //     because that will extract '/**/' as a path parameter, which a user never specified.
496                 return new PrefixAddingPathMapping(pathPrefix, mapping);
497             }
498         }
499 
500         if (pattern.startsWith(REGEX)) {
501             return new PrefixAddingPathMapping(pathPrefix, mapping);
502         }
503 
504         if (pattern.startsWith("/")) {
505             // Default pattern
506             return PathMapping.of(concatPaths(pathPrefix, pattern));
507         }
508 
509         // Should never reach here because we validated the path pattern.
510         throw new Error();
511     }

513     /**
514      * Returns a specified path pattern. The path pattern might be specified by {@link Path} or
515      * HTTP method annotations such as {@link Get} and {@link Post}.
516      */
517     private static String findPattern(Method method, Set<Annotation> methodAnnotations) {
518         String pattern = findAnnotation(method, Path.class).map(Path::value)
519                                                            .orElse(null);
520         for (Annotation a : methodAnnotations) {
521             final String p = (String) invokeValueMethod(a);
522             if (DefaultValues.isUnspecified(p)) {
523                 continue;
524             }
525             checkArgument(pattern == null,
526                           "Only one path can be specified. (" + pattern + ", " + p + ')');
527             pattern = p;
528         }
529         if (pattern == null || pattern.isEmpty()) {
530             throw new IllegalArgumentException(
531                     "A path pattern should be specified by @Path or HTTP method annotations.");
532         }
533         return pattern;
534     }

536     /**
537      * Returns a decorator chain which is specified by {@link Decorator} annotations and user-defined
538      * decorator annotations.
539      */
540     private static Function<Service<HttpRequest, HttpResponse>,
541             ? extends Service<HttpRequest, HttpResponse>> decorator(
542             Method method, Class<?> clazz,
543             Function<Service<HttpRequest, HttpResponse>,
544                     ? extends Service<HttpRequest, HttpResponse>> initialDecorator) {
545 
546         final List<DecoratorAndOrder> decorators = collectDecorators(clazz, method);
547 
548         Function<Service<HttpRequest, HttpResponse>,
549                 ? extends Service<HttpRequest, HttpResponse>> decorator = initialDecorator;
550         for (int i = decorators.size() - 1; i >= 0; i--) {
551             final DecoratorAndOrder d = decorators.get(i);
552             decorator = decorator.andThen(d.decorator());
553         }
554         return decorator;
555     }

557     /**
558      * Returns a decorator list which is specified by {@link Decorator} annotations and user-defined
559      * decorator annotations.
560      */
561     @VisibleForTesting
562     static List<DecoratorAndOrder> collectDecorators(Class<?> clazz, Method method) {
563         final List<DecoratorAndOrder> decorators = new ArrayList<>();
564 
565         // Class-level decorators are applied before method-level decorators.
566         collectDecorators(decorators, clazz.getAnnotations());
567         collectDecorators(decorators, method.getAnnotations());
568 
569         // Sort decorators by "order" attribute values.
570         decorators.sort(Comparator.comparing(DecoratorAndOrder::order));
571 
572         return decorators;
573     }

575     /**
576      * Adds decorators to the specified {@code list}. Decorators which are annotated with {@link Decorator}
577      * and user-defined decorators will be collected.
578      */
579     private static void collectDecorators(List<DecoratorAndOrder> list, Annotation[] annotations) {
580         if (annotations.length == 0) {
581             return;
582         }
583 
584         // Respect the order of decorators which is specified by a user. The first one is first applied
585         // for most of the cases. But if @Decorator and user-defined decorators are specified in a mixed order,
586         // the specified order and the applied order can be different. To overcome this problem, we introduce
587         // "order" attribute to @Decorator annotation to sort decorators. If a user-defined decorator
588         // annotation has "order" attribute, it will be also used for sorting.
589         for (final Annotation annotation : annotations) {
590             if (annotation instanceof Decorator) {
591                 final Decorator d = (Decorator) annotation;
592                 list.add(new DecoratorAndOrder(d, newDecorator(d), d.order()));
593                 continue;
594             }
595 
596             if (annotation instanceof Decorators) {
597                 final Decorator[] decorators = ((Decorators) annotation).value();
598                 for (final Decorator d : decorators) {
599                     list.add(new DecoratorAndOrder(d, newDecorator(d), d.order()));
600                 }
601                 continue;
602             }
603 
604             DecoratorAndOrder udd = userDefinedDecorator(annotation);
605             if (udd != null) {
606                 list.add(udd);
607                 continue;
608             }
609 
610             // If user-defined decorators are repeatable and they are specified more than once.
611             try {
612                 final Method method = Iterables.getFirst(getMethods(annotation.annotationType(),
613                                                                     withName("value")), null);
614                 assert method != null : "No 'value' method is found from " + annotation;
615                 final Annotation[] decorators = (Annotation[]) method.invoke(annotation);
616                 for (final Annotation decorator : decorators) {
617                     udd = userDefinedDecorator(decorator);
618                     if (udd == null) {
619                         break;
620                     }
621                     list.add(udd);
622                 }
623             } catch (Throwable ignore) {
624                 // The annotation may be a container of a decorator or may be not, so we just ignore
625                 // any exception from this clause.
626             }
627         }
628     }
629 
630     /**
631      * Returns a decorator with its order if the specified {@code annotation} is one of the user-defined
632      * decorator annotation.
633      */
634     @Nullable
635     private static DecoratorAndOrder userDefinedDecorator(Annotation annotation) {
636         // User-defined decorator MUST be annotated with @DecoratorFactory annotation.
637         final DecoratorFactory d =
638                 findAnnotation(annotation.annotationType(), DecoratorFactory.class).orElse(null);
639         if (d == null) {
640             return null;
641         }
642 
643         // In case of user-defined decorator, we need to create a new decorator from its factory.
644         @SuppressWarnings("unchecked")
645         final DecoratorFactoryFunction<Annotation> factory = getInstance(d, DecoratorFactoryFunction.class);
646 
647         // If the annotation has "order" attribute, we can use it when sorting decorators.
648         int order = 0;
649         try {
650             final Method method = Iterables.getFirst(getMethods(annotation.annotationType(),
651                                                                 withName("order")), null);
652             if (method != null) {
653                 final Object value = method.invoke(annotation);
654                 if (value instanceof Integer) {
655                     order = (Integer) value;
656                 }
657             }
658         } catch (Throwable ignore) {
659             // A user-defined decorator may not have an 'order' attribute.
660             // If it does not exist, '0' is used by default.
661         }
662         return new DecoratorAndOrder(annotation, factory.newDecorator(annotation), order);
663     }

702     /**
703      * Returns an immutable list builder of objects which are specified as the {@code value} of annotation
704      * {@code T}.
705      */
706     private static <T extends Annotation, R> Builder<R> annotationValues(
707             Method targetMethod, Class<?> targetClass, Class<T> annotationClass, Class<?> repeatableClass,
708             Class<R> expectedType) {
709 
710         requireNonNull(annotationClass, "annotationClass");
711         requireNonNull(targetMethod, "targetMethod");
712         requireNonNull(targetClass, "targetClass");
713 
714         final Builder<R> builder = new Builder<>();
715         annotationValues0(builder, targetMethod, annotationClass, repeatableClass, expectedType);
716         getClasses(targetClass, new Builder<>()).forEach(
717                 target -> annotationValues0(builder, target, annotationClass, repeatableClass, expectedType));
718         return builder;
719     }
720 
721     private static <T extends Annotation, R> void annotationValues0(
722             Builder<R> builder, AnnotatedElement target, Class<T> annotationClass, Class<?> repeatableClass,
723             Class<R> expectedType) {
724         for (final Annotation annotation : target.getAnnotations()) {
725             final Class<? extends Annotation> type = annotation.annotationType();
726             if (type == annotationClass) {
727                 builder.add(getInstance(annotation, expectedType));
728             } else if (type == repeatableClass) {
729                 @SuppressWarnings("unchecked")
730                 final T[] annotations = (T[]) invokeValueMethod(annotation);
731                 Arrays.stream(annotations).forEach(a -> builder.add(getInstance(a, expectedType)));
732             }
733         }
734     }
735 
736     private static List<Class<?>> getClasses(Class<?> clazz, Builder<Class<?>> builder) {
737         final Class<?> superClass = clazz.getSuperclass();
738         if (superClass == null || superClass == Object.class) {
739             builder.add(clazz);
740             return builder.build();
741         }
742 
743         // Subclass first.
744         builder.add(clazz);
745         return getClasses(superClass, builder);
746     }

748     /**
749      * Returns a cached instance of the specified {@link Class} which is specified in the given
750      * {@link Annotation}.
751      */
752     static <T> T getInstance(Annotation annotation, Class<T> expectedType) {
753         try {
754             @SuppressWarnings("unchecked")
755             final Class<? extends T> clazz = (Class<? extends T>) invokeValueMethod(annotation);
756             return expectedType.cast(instanceCache.computeIfAbsent(clazz, type -> {
757                 try {
758 
759                     final Constructor<? extends T> constructor =
760                             Iterables.getFirst(getConstructors(clazz, withParametersCount(0)), null);
761                     assert constructor != null : "No default constructor is found from " + clazz.getName();
762                     constructor.setAccessible(true);
763                     return constructor.newInstance();
764                 } catch (Exception e) {
765                     throw new IllegalStateException(
766                             "A class specified in @" + annotation.getClass().getSimpleName() +
767                             " annotation must have an accessible default constructor: " + clazz.getName(), e);
768                 }
769             }));
770         } catch (ClassCastException e) {
771             throw new IllegalArgumentException(
772                     "A class specified in @" + annotation.getClass().getSimpleName() +
773                     " annotation cannot be cast to " + expectedType, e);
774         }
775     }

778      * Returns a cached instance of the specified {@link Class}.
779      */
780     @SuppressWarnings("unchecked")
781     static <T> T getInstance(Class<T> clazz) {
782         return (T) instanceCache.computeIfAbsent(clazz, type -> {
783             try {
784                 final Constructor<? extends T> constructor =
785                         Iterables.getFirst(getConstructors(clazz, withParametersCount(0)), null);
786                 assert constructor != null : "No default constructor is found from " + clazz.getName();
787                 constructor.setAccessible(true);
788                 return constructor.newInstance();
789             } catch (Exception e) {
790                 throw new IllegalStateException("A class must have an accessible default constructor: " +
791                                                 clazz.getName(), e);
792             }
793         });
794     }

797      * Returns an object which is returned by {@code value()} method of the specified annotation {@code a}.
798      */
799     private static Object invokeValueMethod(Annotation a) {
800         try {
801             final Method method = Iterables.getFirst(getMethods(a.getClass(), withName("value")), null);
802             assert method != null : "No 'value' method is found from " + a;
803             return method.invoke(a);
804         } catch (Exception e) {
805             throw new IllegalStateException("An annotation @" + a.getClass().getSimpleName() +
806                                             " must have a 'value' method", e);
807         }

813     @Nullable
814     static String findDescription(AnnotatedElement annotatedElement) {
815         requireNonNull(annotatedElement, "annotatedElement");
816         final Optional<Description> description = findAnnotation(annotatedElement, Description.class);
817         if (description.isPresent()) {
818             final String value = description.get().value();
819             if (DefaultValues.isSpecified(value)) {
820                 checkArgument(!value.isEmpty(), "value is empty");

851     /**
852      * A {@link PathMapping} implementation that combines path prefix and another {@link PathMapping}.
853      */
854     @VisibleForTesting
855     static final class PrefixAddingPathMapping extends AbstractPathMapping {
856 
857         private final String pathPrefix;
858         private final PathMapping mapping;
859         private final String loggerName;
860         private final String meterTag;
861 
862         PrefixAddingPathMapping(String pathPrefix, PathMapping mapping) {
863             requireNonNull(mapping, "mapping");
864             // mapping should be GlobPathMapping or RegexPathMapping
865             assert mapping.regex().isPresent() : "unexpected mapping type: " + mapping.getClass().getName();
866             this.pathPrefix = requireNonNull(pathPrefix, "pathPrefix");
867             this.mapping = mapping;
868             loggerName = newLoggerName(pathPrefix) + '.' + mapping.loggerName();
869             meterTag = PREFIX + pathPrefix + ',' + mapping.meterTag();
870         }
871 
872         @Override
873         protected PathMappingResult doApply(PathMappingContext mappingCtx) {
874             final String path = mappingCtx.path();
875             if (!path.startsWith(pathPrefix)) {
876                 return PathMappingResult.empty();
877             }
878 
879             final PathMappingResult result =
880                     mapping.apply(mappingCtx.overridePath(path.substring(pathPrefix.length() - 1)));
881             if (result.isPresent()) {
882                 return PathMappingResult.of(path, mappingCtx.query(), result.pathParams());
883             } else {
884                 return PathMappingResult.empty();
885             }
886         }
887 
888         @Override
889         public Set<String> paramNames() {
890             return mapping.paramNames();
891         }
892 
893         @Override
894         public String loggerName() {
895             return loggerName;
896         }
897 
898         @Override
899         public String meterTag() {
900             return meterTag;
901         }
902 
903         @Override
904         public Optional<String> prefix() {
905             return Optional.of(pathPrefix);
906         }
907 
908         @Override
909         public Optional<String> regex() {
910             return mapping.regex();
911         }
912 
913         @Override
914         public boolean equals(Object o) {
915             if (this == o) {
916                 return true;
917             }
918             if (!(o instanceof PrefixAddingPathMapping)) {
919                 return false;
920             }
921 
922             final PrefixAddingPathMapping that = (PrefixAddingPathMapping) o;
923             return pathPrefix.equals(that.pathPrefix) && mapping.equals(that.mapping);
924         }
925 
926         @Override
927         public int hashCode() {
928             return 31 * pathPrefix.hashCode() + mapping.hashCode();
929         }
930 
931         @Override
932         public String toString() {
933             return '[' + PREFIX + pathPrefix + ", " + mapping + ']';
934         }
935     }
